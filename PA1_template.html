Create a new dataset equal to the original dataset but with the NAs filled in.

new<-old
for(i in 1:dim(old)[1]){
     if(is.na(old$steps[i])){
          new$steps[i]<-as.integer(oldAvg[
               as.integer(oldAvg[,1])==as.integer(old$interval[i]),2])
     }
}
